// 널병합 연산자
|| 연산자 -> false 값일 때 뒤로 넘어감
&& 연산자 -> true 값일 때 뒤로 넘어감
?? 연산자(널병합 연산자) -> null, undefined 값일 때 뒤로 넘어감

// 옵셔널 체이닝 연산자
// 자주 발생하는 에러
const c = {};
c.b; -> 이건 오류 안남
const a = null;
a.d; -> 이럴 때 오류가 나옴(값이 null이니까)
그래서 바꿀려면
if(a){
    a.d; -> 이렇게 해주어야함
}
ㄴ 이걸 ->              a?.d;(옵셔널 체이닝)로 해주게 됨
ㄴ 배열이나 객체는 ->    a?.[0];
